cyclone_raw <- read_csv(paste0(base_dir,"cyclone_tracks.csv"))
cyclone <- cyclone_raw |> drop_na(TM) |> drop_na(MAX_WIND_SPD)
# cyclone |> summary()
cyclone <- cyclone |> 
  filter(!NAME %in% c("unnamed","noname","UNNAMED","Unnamed")) |> 
  filter(year(TM) > 1973) |> 
  mutate(NAME = str_to_title(NAME)) |> 
  select(c("NAME","TM","LAT", "LON", "MAX_WIND_SPD","CENTRAL_PRES")) |> 
  mutate(MAX_WIND_SPD = round(MAX_WIND_SPD * 3.6,0)) |> 
  rename(name=NAME, 
         date_time = TM, 
         lat = LAT, 
         lon = LON, 
         max_wind_speed = MAX_WIND_SPD,
         central_pressure = "CENTRAL_PRES")

origin_basin <- function(lon) {
  case_when(
    (lon<114) ~ "Indian",  #       Indian <= 114
    (lon>142) ~ "Pacific", # 114 <= Timor <= 142
    TRUE      ~ "Timor"    #      Pacific >  142
  )
}

cyclone <- cyclone |>
  group_by(name) |>
  nest() |>
  mutate(
    origin = map_chr(data,\(data) {
      origin_basin(data$lon[1]) #origin basin from first lon
      })
  ) |>
  unnest(cols = c(data))


cyclone <- cyclone |> 
  mutate(Year = format(date_time, "%Y"), Month = format(date_time, "%m"), 
  origin = as_factor(origin))

cyclone |> write_csv("cyclone.csv")

cyclone_monthly <- cyclone |>
  mutate(YearMonth = floor_date(date_time,unit = "month") )|>
  group_by(YearMonth) |>
  summarise(Month_Freq = n_distinct(name),
            Avg_Wind = round(mean(max_wind_speed, na.rm = TRUE),0),
            Max_Wind = round(max(max_wind_speed)),0) |>
  ungroup()

SOI_df <- SOI_raw |>
  mutate(YearMonth=ym(X1), Month_SOI = X2) |> 
  filter(between(YearMonth, as.Date('1974-01-01'), as.Date('2022-03-31'))) |> 
  select(-c(X1, X2))

SOI_df <- SOI_df |> 
  group_by(year(YearMonth)) |> 
  mutate(Year_SOI = mean(Month_SOI, na.rm = TRUE)) |> 
  ungroup() |> 
  select(-c("year(YearMonth)"))

cyclone_monthly <- 
  left_join(SOI_df, cyclone_monthly, by="YearMonth")
